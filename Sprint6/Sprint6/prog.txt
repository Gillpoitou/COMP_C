int diff(int x, int y){
      return x - y;     
}

int min(int x, int y){
      if(x < y){
            return x;
      } else {
            return y;
      }
}

int fact(int x){
      int res = 1;
      if(x != 0){
            res = fact(x - 1) * x;
      }
      return res;
}
